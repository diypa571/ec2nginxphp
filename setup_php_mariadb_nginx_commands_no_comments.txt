Alt 1
server {
    listen 80;
    server_name your_domain_or_ip;
    root /var/www/html;

    index index.php index.html index.htm;

    location / {
        try_files $uri $uri/ =404;
    }

    location ~ \.php$ {
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;
    }

    location ~ /\.ht {
        deny all;
    }
}

Alt 3

server {
    listen 80;
    server_name your_domain_or_ip;
    root /var/www/html;

    index index.php index.html index.htm;

    # Redirecting any .php?id=xxx to /path/xxx
    location ~ ^/(.*)\.php\?id=(\d+)$ {
        return 301 /$1/$2;
    }

    location / {
        try_files $uri $uri/ =404;
    }

    location ~ \.php$ {
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;
    }

    location ~ /\.ht {
        deny all;
    }
}
